```javascript
import { useState, useEffect } from 'react'
import { useParams, useNavigate, Link } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'
import { caregivers } from '../data/mockData'

// Commissione KYD: 15% sul totale
const KYD_COMMISSION_RATE = 0.15

// Genera numero prenotazione univoco
const generateBookingId = () => {
  const letters = 'ABCDEFGHJKLMNPQRSTUVWXYZ'
  const nums = Math.floor(Math.random() * 90000) + 10000
  const lets = letters[Math.floor(Math.random() * letters.length)] +
               letters[Math.floor(Math.random() * letters.length)]
  return `KYD-${lets}${nums}`
}

// Salva prenotazione in localStorage
const saveBooking = (booking) => {
  const existing = JSON.parse(localStorage.getItem('kyd_bookings') || '[]')
  existing.push(booking)
  localStorage.setItem('kyd_bookings', JSON.stringify(existing))
}

export default function Prenota() {
  const { id } = useParams()
  const navigate = useNavigate()
  const { user } = useAuth()

  const caregiver = caregivers.find(c => c.id === id)

  const [step, setStep] = useState(1)
  const [loading, setLoading] = useState(false)
  const [bookingId, setBookingId] = useState('')

  // STEP 1 — Dati prenotazione
  const [booking, setBooking] = useState({
    date: '',
    startTime: '',
    hours: 2,
    serviceType: caregiver?.services[0] || 'baby',
    notes: '',
    address: ''
  })

  // STEP 3 — Dati pagamento
  const [payment, setPayment] = useState({
    cardNumber: '',
    cardName: '',
    expiry: '',
    cvv: '',
    saveCard: false
  })

  const [errors, setErrors] = useState({})

  // Calcoli prezzi
  const subtotal = caregiver
    ? caregiver.hourlyRate * booking.hours
    : 0
  const kydCommission = Math.round(subtotal * KYD_COMMISSION_RATE * 100) / 100
  const kydCommissionDisplay = kydCommission.toFixed(2)
  const total = (subtotal + kydCommission).toFixed(2)

  // Redirect se non loggato
  useEffect(() => {
    if (!user) {
      navigate('/login')
    }
  }, [user, navigate])

  // Caregiver non trovato
  if (!caregiver) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <span className="text-6xl">😕</span>
          <p className="mt-4 text-gray-600">Caregiver non trovato</p>
          <Link to="/ricerca"
                className="mt-4 inline-block text-blue-600 hover:underline">
            Torna alla ricerca
          </Link>
        </div>
      </div>
    )
  }

  // ─── VALIDAZIONI ────────────────────────────────────────
  const validateStep1 = () => {
    const e = {}
    if (!booking.date) e.date = 'Seleziona una data'
    if (!booking.startTime) e.startTime = 'Seleziona un orario'
    if (!booking.address.trim()) e.address = 'Inserisci l\'indirizzo'
    if (booking.hours < 1) e.hours = 'Minimo 1 ora'
    // Data non nel passato
    if (booking.date && new Date(booking.date) < new Date().setHours(0,0,0,0)) {
      e.date = 'La data non può essere nel passato'
    }
    setErrors(e)
    return Object.keys(e).length === 0
  }

  const validateStep3 = () => {
    const e = {}
    const cardClean = payment.cardNumber.replace(/\s/g, '')
    if (cardClean.length < 16) e.cardNumber = 'Numero carta non valido'
    if (!payment.cardName.trim()) e.cardName = 'Inserisci il nome sul carta'
    if (!payment.expiry.match(/^\d{2}\/\d{2}$/)) {
      e.expiry = 'Formato MM/AA'
    } else {
      const [mm, yy] = payment.expiry.split('/')
      const expDate = new Date(2000 + parseInt(yy), parseInt(mm) - 1)
      if (expDate < new Date()) e.expiry = 'Carta scaduta'
    }
    if (payment.cvv.length < 3) e.cvv = 'CVV non valido'
    setErrors(e)
    return Object.keys(e).length === 0
  }

  // ─── FORMATTER CARTA ────────────────────────────────────
  const formatCardNumber = (val) => {
    const clean = val.replace(/\D/g, '').slice(0, 16)
    return clean.replace(/(.{4})/g, '$1 ').trim()
  }

  const formatExpiry = (val) => {
    const clean = val.replace(/\D/g, '').slice(0, 4)
    if (clean.length >= 2) {
      return clean.slice(0, 2) + '/' + clean.slice(2)
    }
    return clean
  }

  // ─── TIPO CARTA (da numero) ──────────────────────────────
  const getCardType = () => {
    const n = payment.cardNumber.replace(/\s/g, '')
    if (n.startsWith('4')) return { name: 'Visa', emoji: '💳', color: '#1A1F71' }
    if (n.startsWith('5')) return { name: 'Mastercard', emoji: '💳', color: '#EB001B' }
    if (n.startsWith('34') || n.startsWith('37')) {
      return { name: 'Amex', emoji: '💳', color: '#007BC1' }
    }
    return { name: 'Carta', emoji: '💳', color: '#6B7280' }
  }

  // ─── STEP HANDLERS ──────────────────────────────────────
  const goToStep2 = () => {
    if (validateStep1()) setStep(2)
  }

  const goToStep3 = () => {
    setStep(3)
  }

  const handlePayment = async () => {
    if (!validateStep3()) return

    setLoading(true)

    // Simula elaborazione pagamento (1.5s)
    await new Promise(r => setTimeout(r, 1500))

    const id = generateBookingId()
    setBookingId(id)

    // Salva prenotazione
    saveBooking({
      id,
      caregiverId: caregiver.id,
      caregiverName: caregiver.name,
      caregiverPhoto: caregiver.photo,
      userId: user?.id,
      userName: user?.name,
      date: booking.date,
      startTime: booking.startTime,
      hours: booking.hours,
      serviceType: booking.serviceType,
      address: booking.address,
      notes: booking.notes,
      subtotal,
      commission: kydCommission,
      total: parseFloat(total),
      status: 'confermata',
      createdAt: new Date().toISOString()
    })

    setLoading(false)
    setStep(4) // Conferma
  }

  // ─── SERVICE LABELS ─────────────────────────────────────
  const serviceLabels = {
    baby: '👶 Babysitting',
    pet: '🐾 Pet Sitting',
    elder: '👴 Assistenza Anziani'
  }

  // ─── MIN DATE (oggi) ─────────────────────────────────────
  const today = new Date().toISOString().split('T')[0]

  // ─── PROGRESS BAR ───────────────────────────────────────
  const steps = [
    { n: 1, label: 'Dettagli' },
    { n: 2, label: 'Riepilogo' },
    { n: 3, label: 'Pagamento' },
    { n: 4, label: 'Conferma' }
  ]

  return (
    <div className="min-h-screen bg-gray-50">

      {/* NAVBAR MINIMALE */}
      <nav className="bg-white border-b px-6 py-4 flex items-center
                      justify-between sticky top-0 z-40 shadow-sm">
        <Link to="/" className="flex items-center gap-2">
          <span className="text-2xl">🛡️</span>
          <span className="font-bold text-blue-600 text-xl">KYD</span>
        </Link>
        <div className="text-sm text-gray-500">
          🔒 Pagamento sicuro — SSL/TLS 256bit
        </div>
      </nav>

      <div className="max-w-2xl mx-auto px-4 py-8">

        {/* PROGRESS BAR */}
        {step < 4 && (
          <div className="mb-8">
            <div className="flex items-center justify-between mb-2">
              {steps.filter(s => s.n < 4).map((s, i) => (
                <div key={s.n} className="flex items-center flex-1">
                  <div className="flex flex-col items-center">
                    <div className={`w-9 h-9 rounded-full flex items-center
                                    justify-center text-sm font-bold
                                    transition-all duration-300 ${
                      step > s.n
                        ? 'bg-green-500 text-white'
                        : step === s.n
                          ? 'bg-blue-600 text-white shadow-lg shadow-blue-200'
                          : 'bg-gray-200 text-gray-500'
                    }`}>
                      {step > s.n ? '✓' : s.n}
                    </div>
                    <span className={`text-xs mt-1 font-medium ${
                      step >= s.n ? 'text-blue-600' : 'text-gray-400'
                    }`}>
                      {s.label}
                    </span>
                  </div>
                  {i < 2 && (
                    <div className={`flex-1 h-1 mx-2 mb-5 rounded transition-all
                                    duration-500 ${
                      step > s.n ? 'bg-green-400' : 'bg-gray-200'
                    }`}/>
                  )}
                </div>
              ))}
            </div>
          </div>
        )}

        {/* ════════════════════════════════════════════
            STEP 1 — DETTAGLI PRENOTAZIONE
        ════════════════════════════════════════════ */}
        {step === 1 && (
          <div className="space-y-6">

            {/* Card caregiver selezionato */}
            <div className="bg-white rounded-2xl border border-gray-100
                            shadow-sm p-5">
              <div className="flex items-center gap-4">
                <img
                  src={caregiver.photo}
                  alt={caregiver.name}
                  className="w-16 h-16 rounded-xl object-cover"
                  onError={e => {
                    e.target.style.display = 'none'
                  }}
                />
                <div>
                  <div className="flex items-center gap-2">
                    <h2 className="font-bold text-gray-900 text-lg">
                      {caregiver.name}
                    </h2>
                    <span className="text-blue-500">✅</span>
                  </div>
                  <div className="text-sm text-gray-500">
                    📍 {caregiver.neighborhood}, {caregiver.city}
                  </div>
                  <div className="flex items-center gap-3 mt-1">
                    <span className="text-amber-400 text-sm">
                      {'★'.repeat(Math.floor(caregiver.rating))}
                    </span>
                    <span className="text-sm font-semibold text-gray-700">
                      {caregiver.rating}
                    </span>
                    <span className="font-bold text-blue-600">
                      €{caregiver.hourlyRate}/ora
                    </span>
                  </div>
                </div>
              </div>
            </div>

            {/* Form dettagli */}
            <div className="bg-white rounded-2xl border border-gray-100
                            shadow-sm p-6 space-y-5">
              <h3 className="font-bold text-gray-900 text-lg">
                📅 Dettagli della prenotazione
              </h3>

              {/* Tipo servizio (se il caregiver ne offre più di uno) */}
              {caregiver.services.length > 1 && (
                <div>
                  <label className="block text-sm font-semibold text-gray-700 mb-2">
                    Tipo di servizio
                  </label>
                  <div className="grid grid-cols-2 gap-2">
                    {caregiver.services.map(s => (
                      <button
                        key={s}
                        onClick={() => setBooking(b => ({...b, serviceType: s}))}
                        className={`py-2.5 px-4 rounded-xl border-2 text-sm
                                    font-medium transition-all ${
                          booking.serviceType === s
                            ? 'border-blue-500 bg-blue-50 text-blue-700'
                            : 'border-gray-200 text-gray-600 hover:border-blue-300'
                        }`}
                      >
                        {serviceLabels[s]}
                      </button>
                    ))}
                  </div>
                </div>
              )}

              {/* Data */}
              <div>
                <label className="block text-sm font-semibold text-gray-700 mb-1.5">
                  Data *
                </label>
                <input
                  type="date"
                  min={today}
                  value={booking.date}
                  onChange={e => setBooking(b => ({...b, date: e.target.value}))}
                  className={`w-full border-2 rounded-xl px-4 py-3 text-sm
                              focus:outline-none transition ${
                    errors.date
                      ? 'border-red-400 bg-red-50'
                      : 'border-gray-200 focus:border-blue-500'
                  }`}
                />
                {errors.date && (
                  <p className="text-red-500 text-xs mt-1">⚠ {errors.date}</p>
                )}
              </div>

              {/* Orario inizio + Numero ore */}
              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-semibold text-gray-700 mb-1.5">
                    Orario inizio *
                  </label>
                  <select
                    value={booking.startTime}
                    onChange={e => setBooking(b => ({
                      ...b, startTime: e.target.value
                    }))}
                    className={`w-full border-2 rounded-xl px-4 py-3 text-sm
                                focus:outline-none bg-white transition ${
                      errors.startTime
                        ? 'border-red-400'
                        : 'border-gray-200 focus:border-blue-500'
                    }`}
                  >
                    <option value="">Seleziona</option>
                    {Array.from({length: 24}, (_, i) => {
                      const h = String(i).padStart(2, '0')
                      return (
                        <option key={h} value={`${h}:00`}>{h}:00</option>
                      )
                    })}
                  </select>
                  {errors.startTime && (
                    <p className="text-red-500 text-xs mt-1">
                      ⚠ {errors.startTime}
                    </p>
                  )}
                </div>

                <div>
                  <label className="block text-sm font-semibold text-gray-700 mb-1.5">
                    Durata (ore)
                  </label>
                  <div className="flex items-center border-2 border-gray-200
                                  rounded-xl overflow-hidden focus-within:border-blue-500">
                    <button
                      onClick={() => setBooking(b => ({
                        ...b, hours: Math.max(1, b.hours - 1)
                      }))}
                      className="px-4 py-3 text-gray-500 hover:bg-gray-50
                                 text-xl font-bold transition"
                    >−</button>
                    <div className="flex-1 text-center font-bold text-gray-900 text-lg">
                      {booking.hours}h
                    </div>
                    <button
                      onClick={() => setBooking(b => ({
                        ...b, hours: Math.min(12, b.hours + 1)
                      }))}
                      className="px-4 py-3 text-gray-500 hover:bg-gray-50
                                 text-xl font-bold transition"
                    >+</button>
                  </div>
                </div>
              </div>

              {/* Indirizzo */}
              <div>
                <label className="block text-sm font-semibold text-gray-700 mb-1.5">
                  Indirizzo del servizio *
                </label>
                <input
                  type="text"
                  value={booking.address}
                  onChange={e => setBooking(b => ({
                    ...b, address: e.target.value
                  }))}
                  placeholder="Via Roma 1, Milano"
                  className={`w-full border-2 rounded-xl px-4 py-3 text-sm
                              focus:outline-none transition ${
                    errors.address
                      ? 'border-red-400 bg-red-50'
                      : 'border-gray-200 focus:border-blue-500'
                  }`}
                />
                {errors.address && (
                  <p className="text-red-500 text-xs mt-1">⚠ {errors.address}</p>
                )}
              </div>

              {/* Note */}
              <div>
                <label className="block text-sm font-semibold text-gray-700 mb-1.5">
                  Note per il caregiver
                  <span className="text-gray-400 font-normal ml-1">(opzionale)</span>
                </label>
                <textarea
                  value={booking.notes}
                  onChange={e => setBooking(b => ({
                    ...b, notes: e.target.value
                  }))}
                  placeholder="Es: mia figlia ha 3 anni, ama disegnare. Il cane si chiama Rocky..."
                  rows={3}
                  className="w-full border-2 border-gray-200 rounded-xl px-4 py-3
                             text-sm focus:outline-none focus:border-blue-500
                             resize-none transition"
                />
              </div>
            </div>

            {/* Preview prezzo */}
            <div className="bg-blue-50 border border-blue-200 rounded-2xl p-4
                            flex items-center justify-between">
              <div>
                <div className="text-sm text-blue-700">Stima costo totale</div>
                <div className="text-xs text-blue-500 mt-0.5">
                  {booking.hours}h × €{caregiver.hourlyRate} + commissione KYD
                </div>
              </div>
              <div className="text-2xl font-bold text-blue-700">
                €{total}
              </div>
            </div>

            <button
              onClick={goToStep2}
              className="w-full bg-blue-600 hover:bg-blue-700 text-white
                         font-bold py-4 rounded-2xl transition-all duration-200
                         text-lg shadow-lg shadow-blue-200"
            >
              Continua → Riepilogo
            </button>
          </div>
        )}

        {/* ════════════════════════════════════════════
            STEP 2 — RIEPILOGO ORDINE
        ════════════════════════════════════════════ */}
        {step === 2 && (
          <div className="space-y-6">
            <div className="bg-white rounded-2xl border border-gray-100
                            shadow-sm p-6">
              <h3 className="font-bold text-gray-900 text-xl mb-6">
                📋 Riepilogo prenotazione
              </h3>

              {/* Caregiver */}
              <div className="flex items-center gap-4 p-4 bg-gray-50
                              rounded-xl mb-6">
                <img
                  src={caregiver.photo}
                  alt={caregiver.name}
                  className="w-14 h-14 rounded-xl object-cover"
                />
                <div>
                  <div className="font-bold text-gray-900">{caregiver.name}</div>
                  <div className="text-sm text-gray-500">
                    {serviceLabels[booking.serviceType]}
                  </div>
                  <div className="text-sm text-gray-500">
                    📍 {booking.address}
                  </div>
                </div>
              </div>

              {/* Dettagli servizio */}
              <div className="space-y-3 mb-6">
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">📅 Data</span>
                  <span className="font-semibold text-gray-900">
                    {new Date(booking.date).toLocaleDateString('it-IT', {
                      weekday: 'long',
                      day: 'numeric',
                      month: 'long',
                      year: 'numeric'
                    })}
                  </span>
                </div>
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">🕐 Orario</span>
                  <span className="font-semibold text-gray-900">
                    {booking.startTime} —{' '}
                    {String(
                      (parseInt(booking.startTime) + booking.hours) % 24
                    ).padStart(2,'0')}:00
                  </span>
                </div>
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">⏱ Durata</span>
                  <span className="font-semibold text-gray-900">
                    {booking.hours} {booking.hours === 1 ? 'ora' : 'ore'}
                  </span>
                </div>
                {booking.notes && (
                  <div className="flex justify-between text-sm">
                    <span className="text-gray-600">📝 Note</span>
                    <span className="font-semibold text-gray-900 text-right
                                     max-w-48">
                      {booking.notes}
                    </span>
                  </div>
                )}
              </div>

              {/* Separatore */}
              <div className="border-t border-dashed border-gray-200 my-4"/>

              {/* ─── CALCOLO PREZZI ─── */}
              <div className="space-y-3">

                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">
                    Tariffa caregiver
                  </span>
                  <span className="text-gray-900">
                    {booking.hours}h × €{caregiver.hourlyRate}/h
                  </span>
                </div>
                <div className="flex justify-between text-sm">
                  <span className="text-gray-900 font-semibold">
                    Subtotale
                  </span>
                  <span className="font-semibold text-gray-900">
                    €{subtotal.toFixed(2)}
                  </span>
                </div>

                {/* Commissione KYD */}
                <div className="flex justify-between text-sm items-start">
                  <div>
                    <span className="text-gray-600">
                      Commissione KYD (15%)
                    </span>
                    <div className="text-xs text-gray-400 mt-0.5 max-w-52">
                      Copre verifica caregiver, assicurazione RC,
                      supporto 24/7 e garanzia rimborso
                    </div>
                  </div>
                  <span className="text-gray-700 font-medium ml-4">
                    +€{kydCommissionDisplay}
                  </span>
                </div>

                {/* Separatore */}
                <div className="border-t border-gray-200 my-2"/>

                {/* TOTALE */}
                <div className="flex justify-between items-center">
                  <span className="text-lg font-bold text-gray-900">
                    Totale da pagare
                  </span>
                  <span className="text-2xl font-bold text-blue-600">
                    €{total}
                  </span>
                </div>

                {/* Breakdown payout */}
                <div className="bg-gray-50 rounded-xl p-4 space-y-2 mt-2">
                  <div className="text-xs font-semibold text-gray-500
                                   uppercase tracking-wide mb-2">
                    Come vengono usati i tuoi €{total}
                  </div>
                  <div className="flex justify-between text-xs">
                    <span className="flex items-center gap-2">
                      <span className="w-3 h-3 bg-green-500 rounded-full"/>
                      Pagato a {caregiver.name}
                    </span>
                    <span className="font-semibold text-green-600">
                      €{subtotal.toFixed(2)}
                    </span>
                  </div>
                  <div className="flex justify-between text-xs">
                    <span className="flex items-center gap-2">
                      <span className="w-3 h-3 bg-blue-500 rounded-full"/>
                      Commissione KYD
                    </span>
                    <span className="font-semibold text-blue-600">
                      €{kydCommissionDisplay}
                    </span>
                  </div>
                  {/* Barra visiva split */}
                  <div className="flex rounded-full overflow-hidden h-2 mt-2">
                    <div
                      className="bg-green-500 transition-all"
                      style={{ width: `${(1 - KYD_COMMISSION_RATE) * 100}%` }}
                    />
                    <div
                      className="bg-blue-500"
                      style={{ width: `${KYD_COMMISSION_RATE * 100}%` }}
                    />
                  </div>
                </div>
              </div>
            </div>

            {/* Garanzie */}
            <div className="grid grid-cols-3 gap-3">
              {[
                { icon: '🔒', text: 'Pagamento sicuro Stripe' },
                { icon: '↩️', text: 'Rimborso gratuito -24h' },
                { icon: '🛡️', text: 'RC inclusa nel servizio' }
              ].map((g, i) => (
                <div key={i} className="bg-white rounded-xl border p-3
                                         text-center text-xs text-gray-600
                                         shadow-sm">
                  <div className="text-2xl mb-1">{g.icon}</div>
                  {g.text}
                </div>
              ))}
            </div>

            <div className="flex gap-3">
              <button
                onClick={() => setStep(1)}
                className="flex-1 border-2 border-gray-200 text-gray-600
                           hover:border-gray-300 font-semibold py-3 rounded-2xl
                           transition"
              >
                ← Modifica
              </button>
              <button
                onClick={goToStep3}
                className="flex-2 bg-blue-600 hover:bg-blue-700 text-white
                           font-bold py-3 px-8 rounded-2xl transition
                           shadow-lg shadow-blue-200"
              >
                Vai al pagamento →
              </button>
            </div>
          </div>
        )}

        {/* ════════════════════════════════════════════

        ════════════════════════════════════════════ */}
```

Crea questo hook per unire caregiver mockData + custom:

```javascript
// src/hooks/useCaregivers.js
import { useState, useEffect } from 'react'
import { caregivers as mockCaregivers } from '../data/mockData'

export function useCaregivers() {
  const [allCaregivers, setAllCaregivers] = useState(mockCaregivers)

  useEffect(() => {
    // Carica caregiver custom da localStorage
    const loadCustom = () => {
      const custom = JSON.parse(
        localStorage.getItem('kyd_custom_caregivers') || '[]'
      )
      setAllCaregivers([...mockCaregivers, ...custom])
    }

    loadCustom()

    // Aggiorna quando cambia localStorage
    window.addEventListener('storage', loadCustom)
    return () => window.removeEventListener('storage', loadCustom)
  }, [])

  return allCaregivers
}
```
