import { createContext, useContext, useState, useEffect } from 'react'

const AuthContext = createContext(null)

// Database utenti simulato (localStorage)
const USERS_KEY = 'kyd_users'
const SESSION_KEY = 'kyd_session'

// Utenti demo pre-caricati
const DEMO_USERS = [
  {
    id: 'u1',
    email: 'famiglia@demo.it',
    password: 'demo123',
    name: 'Marco Rossi',
    role: 'family',
    avatar: '👨‍👩‍👧',
    city: 'Milano',
    createdAt: '2024-01-01'
  },
  {
    id: 'u2',
    email: 'caregiver@demo.it',
    password: 'demo123',
    name: 'Sara Martinelli',
    role: 'caregiver',
    avatar: '👩‍⚕️',
    city: 'Milano',
    createdAt: '2024-01-01'
  }
]

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null)
  const [loading, setLoading] = useState(true)

  // Inizializza DB utenti se vuoto
  useEffect(() => {
    const existing = localStorage.getItem(USERS_KEY)
    if (!existing) {
      localStorage.setItem(USERS_KEY, JSON.stringify(DEMO_USERS))
    }

    // Ripristina sessione
    const session = localStorage.getItem(SESSION_KEY)
    if (session) {
      setUser(JSON.parse(session))
    }
    setLoading(false)
  }, [])

  const getUsers = () => {
    return JSON.parse(localStorage.getItem(USERS_KEY) || '[]')
  }

  // LOGIN
  const login = (email, password) => {
    const users = getUsers()
    const found = users.find(
      u => u.email.toLowerCase() === email.toLowerCase()
         && u.password === password
    )
    if (!found) {
      return { success: false, error: 'Email o password errati' }
    }
    const { password: _, ...safeUser } = found
    setUser(safeUser)
    localStorage.setItem(SESSION_KEY, JSON.stringify(safeUser))
    return { success: true, user: safeUser }
  }

  // REGISTRAZIONE
  const register = (data) => {
    const users = getUsers()

    // Check email duplicata
    if (users.find(u => u.email.toLowerCase() === data.email.toLowerCase())) {
      return { success: false, error: 'Email già registrata' }
    }

    const newUser = {
      id: 'u' + Date.now(),
      email: data.email,
      password: data.password,
      name: data.name,
      role: data.role || 'family',
      avatar: data.role === 'caregiver' ? '👩‍⚕️' : '👨‍👩‍👧',
      city: data.city || '',
      createdAt: new Date().toISOString()
    }

    users.push(newUser)
    localStorage.setItem(USERS_KEY, JSON.stringify(users))

    const { password: _, ...safeUser } = newUser
    setUser(safeUser)
    localStorage.setItem(SESSION_KEY, JSON.stringify(safeUser))
    return { success: true, user: safeUser }
  }

  // LOGOUT
  const logout = () => {
    setUser(null)
    localStorage.removeItem(SESSION_KEY)
  }

  return (
    <AuthContext.Provider value={{ user, login, register, logout, loading }}>
      {!loading && children}
    </AuthContext.Provider>
  )
}

export const useAuth = () => useContext(AuthContext)

    name: '', email: '', password: '',
    confirmPassword: '', city: '', acceptTerms: false

    if (!form.email.includes('@')) return 'Email non valida'

    if (form.password !== form.confirmPassword) return 'Le password non coincidono'

    e.preventDefault()
    setError('')

      setError(validationError)

import { useState, useEffect, useRef } from 'react'

// NON importare leaflet direttamente: usa dynamic import

function MappaRicerca({ caregiversFiltrati, selectedId, onSelectCaregiver }) {
  const mapRef = useRef(null)
  const leafletMapRef = useRef(null)
  const markersRef = useRef({})
  const circlesRef = useRef({})

  // Colori per tipo servizio
  const getColor = (caregiver) => {
    return SERVICE_COLORS[caregiver.primaryService] || '#6B7280'
  }

  // Crea SVG marker colorato
  const createMarkerIcon = (color, size = 36) => {
    const L = window.L
    return L.divIcon({
      className: '',
      html: `
        <div style="
          width: ${size}px;
          height: ${size}px;
          background: ${color};
          border: 3px solid white;
          border-radius: 50% 50% 50% 0;
          transform: rotate(-45deg);
          box-shadow: 0 2px 8px rgba(0,0,0,0.3);
          cursor: pointer;
        ">
          <div style="
            transform: rotate(45deg);
            color: white;
            font-size: ${size * 0.4}px;
            display: flex;
            align-items: center;
            justify-content: center;
            height: 100%;
            padding-bottom: 4px;
          ">
            ${caregiversFiltrati.find(c =>
              SERVICE_COLORS[c.primaryService] === color
            )?.primaryService === 'baby' ? '👶' :
              SERVICE_COLORS[Object.keys(SERVICE_COLORS).find(
                k => SERVICE_COLORS[k] === color
              )] === SERVICE_COLORS.pet ? '🐾' : '👴'}
          </div>
        </div>
      `,
      iconSize: [size, size],
      iconAnchor: [size / 2, size],
      popupAnchor: [0, -size]
    })
  }

  // Inizializza mappa
  useEffect(() => {
    if (!mapRef.current || leafletMapRef.current) return
    if (!window.L) {
      console.error('Leaflet non caricato. Aggiungi CDN in index.html')
      return
    }

    const L = window.L
    const map = L.map(mapRef.current, {
      center: [42.5, 12.5], // Centro Italia
      zoom: 6,
      zoomControl: true
    })

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap'
    }).addTo(map)

    leafletMapRef.current = map

    return () => {
      map.remove()
      leafletMapRef.current = null
    }
  }, [])

  // Aggiorna marker quando cambiano i caregiver filtrati
  useEffect(() => {
    const map = leafletMapRef.current
    if (!map || !window.L) return
    const L = window.L

    // Rimuovi marker vecchi
    Object.values(markersRef.current).forEach(m => m.remove())
    Object.values(circlesRef.current).forEach(c => c.remove())

    circlesRef.current = {}

    if (caregiversFiltrati.length === 0) return

      // Cerchio area operativa
      const circle = L.circle(pos, {
        radius: caregiver.radius * 1000,
        fillColor: color,
        fillOpacity: 0.10,
        color: color,
        weight: 2,
        dashArray: '5, 5'
      }).addTo(map)

      circlesRef.current[caregiver.id] = circle

      // Icona emoji servizio
      const emoji = caregiver.primaryService === 'baby' ? '👶'
                  : caregiver.primaryService === 'pet' ? '🐾' : '👴'

      // Marker con icona custom
      const icon = L.divIcon({
        className: '',
        html: `
          <div style="
            position: relative;
            width: 44px;
            height: 44px;
          ">
            <div style="
              width: 44px;
              height: 44px;
              background: ${color};
              border: 3px solid white;
              border-radius: 50%;
              box-shadow: 0 3px 10px rgba(0,0,0,0.35);
              display: flex;
              align-items: center;
              justify-content: center;
              font-size: 20px;
              cursor: pointer;
              transition: transform 0.2s;
            ">${emoji}</div>
            <div style="
              position: absolute;
              bottom: -6px;
              left: 50%;
              transform: translateX(-50%);
              width: 0;
              height: 0;
              border-left: 6px solid transparent;
              border-right: 6px solid transparent;
              border-top: 8px solid ${color};
            "></div>
          </div>
        `,
        iconSize: [44, 52],
        iconAnchor: [22, 52],
        popupAnchor: [0, -52]
      })

      const marker = L.marker(pos, { icon }).addTo(map)

      // Popup al click
      marker.bindPopup(`
        <div style="min-width:200px; font-family: sans-serif;">
          <div style="display:flex; align-items:center; gap:10px; margin-bottom:10px;">

                📍 ${caregiver.city} · ${caregiver.neighborhood}
              </div>
            </div>
          </div>
          <div style="display:flex;gap:6px;flex-wrap:wrap;margin-bottom:8px;">
            ${caregiver.services.map(s => `
              <span style="
                background:${SERVICE_COLORS[s]};
                color:white;
                padding:2px 8px;
                border-radius:20px;
                font-size:11px;
                font-weight:600;
              ">${SERVICE_LABELS[s]}</span>
            `).join('')}
          </div>
          <div style="
            display:flex;
            justify-content:space-between;
            align-items:center;
            margin-bottom:8px;
          ">
            <span style="font-size:13px;">
              ⭐ ${caregiver.rating} (${caregiver.reviewCount})

            <span style="font-weight:700;color:#3B82F6;font-size:15px;">
              €${caregiver.hourlyRate}/h
            </span>
          </div>
          <div style="
            display:inline-block;
            padding:2px 8px;
            border-radius:20px;
            font-size:11px;
            font-weight:600;
            background:${caregiver.available ? '#ECFDF5' : '#F3F4F6'};
            color:${caregiver.available ? '#059669' : '#6B7280'};
            margin-bottom:10px;
          ">
            ${caregiver.available ? '● Disponibile ora' : '○ Non disponibile'}
          </div>
          <div style="display:flex;gap:8px;">

              style="

                padding:8px;

                text-decoration:none;
                text-align:center;
                font-size:12px;

              "
            >Vedi profilo</a>
            <a
              href="/chat?room=${caregiver.id}&user=family"

                flex:1;
                background:#10B981;
                color:white;
                padding:8px;
                border-radius:8px;
                text-decoration:none;
                text-align:center;
                font-size:12px;
                font-weight:600;
              "
            >Chatta</a>
          </div>
        </div>
      `, { maxWidth: 280 })

      marker.on('click', () => {
        onSelectCaregiver(caregiver.id)
      })

      markersRef.current[caregiver.id] = marker
    })

    // Fit bounds per mostrare tutti i marker
    if (caregiversFiltrati.length > 0) {
      const group = L.featureGroup(
        Object.values(markersRef.current)
      )
      map.fitBounds(group.getBounds().pad(0.2), {
        maxZoom: 12
      })
    }
  }, [caregiversFiltrati])

  // Highlight marker selezionato
  useEffect(() => {
    const map = leafletMapRef.current
    if (!map || !window.L) return

    Object.entries(markersRef.current).forEach(([id, marker]) => {
      const caregiver = caregiversFiltrati.find(c => c.id === id)
      if (!caregiver) return
      const color = getColor(caregiver)
      const emoji = caregiver.primaryService === 'baby' ? '👶'
                  : caregiver.primaryService === 'pet' ? '🐾' : '👴'
      const isSelected = id === selectedId
      const size = isSelected ? 54 : 44

      const icon = window.L.divIcon({
        className: '',
        html: `
          <div style="
            position:relative;width:${size}px;height:${size}px;
          ">
            <div style="
              width:${size}px;height:${size}px;
              background:${color};
              border:${isSelected ? '4px solid #1E40AF' : '3px solid white'};
              border-radius:50%;
              box-shadow:${isSelected
                ? '0 0 0 3px rgba(59,130,246,0.4), 0 5px 15px rgba(0,0,0,0.4)'
                : '0 3px 10px rgba(0,0,0,0.35)'};
              display:flex;align-items:center;justify-content:center;
              font-size:${isSelected ? 26 : 20}px;
              cursor:pointer;
            ">${emoji}</div>
            <div style="
              position:absolute;bottom:${isSelected ? -8 : -6}px;
              left:50%;transform:translateX(-50%);
              width:0;height:0;
              border-left:${isSelected ? 8 : 6}px solid transparent;
              border-right:${isSelected ? 8 : 6}px solid transparent;
              border-top:${isSelected ? 10 : 8}px solid ${color};
            "></div>
          </div>
        `,
        iconSize: [size, size + 10],
        iconAnchor: [size / 2, size + 10],
        popupAnchor: [0, -(size + 10)]
      })
      marker.setIcon(icon)

      if (isSelected) {
        marker.openPopup()
      }
    })
  }, [selectedId, caregiversFiltrati])

  return (
    <div
      ref={mapRef}
      style={{
        height: '100%',
        minHeight: '500px',
        width: '100%',
        borderRadius: '0'
      }}
    />
  )

export default function Ricerca() {
  const [searchParams] = useSearchParams()
  const navigate = useNavigate()

  // Filtri con valori iniziali da URL params (se arriva da homepage)
  const [filters, setFilters] = useState({
    service: searchParams.get('servizio') || 'tutti',
    city: searchParams.get('zona') || '',
    maxRate: 30,
    onlyAvailable: false
  })

  const [selectedId, setSelectedId] = useState(null)
  const cardRefs = useRef({})

  // Filtra caregiver
  const caregiversFiltrati = caregivers.filter(c => {
    // Filtro servizio
    if (filters.service !== 'tutti' && !c.services.includes(filters.service)) {
      return false
    }
    // Filtro città (case insensitive, partial match)
    if (filters.city.trim() !== '') {
      const q = filters.city.toLowerCase()
      if (!c.city.toLowerCase().includes(q) &&
          !c.neighborhood.toLowerCase().includes(q)) {
        return false
      }
    }
    // Filtro prezzo
    if (c.hourlyRate > filters.maxRate) return false
    // Filtro disponibilità
    if (filters.onlyAvailable && !c.available) return false

    return true
  })

  // Scroll alla card quando si seleziona un marker
  useEffect(() => {
    if (selectedId && cardRefs.current[selectedId]) {
      cardRefs.current[selectedId].scrollIntoView({
        behavior: 'smooth',
        block: 'center'
      })
    }
  }, [selectedId])

  const getServiceBadgeStyle = (service) => {
    const colors = {
      baby: 'bg-blue-100 text-blue-700',
      pet: 'bg-green-100 text-green-700',
      elder: 'bg-amber-100 text-amber-700'
    }
    return colors[service] || 'bg-gray-100 text-gray-700'
  }

  return (
    <div className="flex h-screen overflow-hidden">

      {/* ─── SIDEBAR SINISTRA ───────────────────── */}
      <div className="w-full md:w-96 flex flex-col bg-white border-r
                      overflow-hidden flex-shrink-0">

        {/* Header sidebar */}
        <div className="p-4 bg-blue-600 text-white flex-shrink-0">
          <div className="flex items-center gap-2 mb-1">
            <span className="text-xl">🔍</span>
            <h1 className="font-bold text-lg">Cerca Caregiver</h1>
          </div>
          <p className="text-blue-100 text-sm">
            {caregiversFiltrati.length} professionisti trovati
          </p>
        </div>

        {/* ─── FILTRI ─────────────────────────── */}
        <div className="p-4 border-b bg-gray-50 flex-shrink-0 space-y-3">

          {/* Tipo servizio */}

            <label className="text-xs font-semibold text-gray-600 uppercase
                               tracking-wide mb-1.5 block">
              Tipo di servizio
            </label>
            <div className="grid grid-cols-4 gap-1">
              {[
                { value:'tutti', label:'Tutti', emoji:'🔍' },
                { value:'baby', label:'Baby', emoji:'👶' },
                { value:'pet', label:'Pet', emoji:'🐾' },
                { value:'elder', label:'Anziani', emoji:'👴' }
              ].map(opt => (
                <button
                  key={opt.value}
                  onClick={() => setFilters(f => ({...f, service: opt.value}))}
                  className={`flex flex-col items-center gap-0.5 py-2 px-1
                              rounded-xl text-xs font-medium transition-all
                              border-2 ${
                    filters.service === opt.value
                      ? 'border-blue-500 bg-blue-50 text-blue-700'
                      : 'border-gray-200 bg-white text-gray-600 hover:border-blue-300'
                  }`}
                >
                  <span className="text-lg">{opt.emoji}</span>
                  <span>{opt.label}</span>
                </button>
              ))}
            </div>
          </div>

          {/* Zona/Città */}
          <div>
            <label className="text-xs font-semibold text-gray-600 uppercase
                               tracking-wide mb-1.5 block">
              Città o quartiere
            </label>
            <input
              type="text"
              value={filters.city}
              onChange={e => setFilters(f => ({...f, city: e.target.value}))}
              placeholder="Es: Milano, Trastevere, Navigli..."
              className="w-full border border-gray-200 rounded-xl px-3 py-2.5
                         text-sm focus:outline-none focus:border-blue-500
                         focus:ring-2 focus:ring-blue-500/20"
            />
          </div>

          {/* Range prezzo */}

              </span>
            </label>
            <input
              type="range"
              min={10}
              max={30}
              value={filters.maxRate}

                ...f, maxRate: parseInt(e.target.value)
              }))}

                Reimposta filtri
              </button>

              {caregiversFiltrati.map(c => (
                <div
                  key={c.id}

                  onClick={() => setSelectedId(
                    selectedId === c.id ? null : c.id
                  )}
                  className={`bg-white rounded-2xl border-2 p-3 cursor-pointer
                              transition-all duration-200 ${
                    selectedId === c.id
