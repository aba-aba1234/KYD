FIX 1 — SCROLL TO TOP AD OGNI PAGINA
════════════════════════════════════════

PROBLEMA:
Quando navigo da /famiglie ad un'altra pagina
e torno su /famiglie, la pagina è scrollata
dove l'avevo lasciata invece di partire dall'inizio.

SOLUZIONE:
Crea il componente ScrollToTop e aggiungilo
all'app. Si attiva automaticamente ad ogni
cambio di route.

Crea src/components/ScrollToTop.jsx:

```jsx
import { useEffect } from 'react'
import { useLocation } from 'react-router-dom'

export default function ScrollToTop() {
  const { pathname } = useLocation()

  useEffect(() => {
    // Ogni volta che cambia la pagina
    // torna automaticamente in cima
    window.scrollTo({
      top: 0,
      left: 0,
      behavior: 'instant' // istantaneo come i siti professionali
    })
  }, [pathname])

  return null // non renderizza niente
}
```

```sql
ALTER TABLE profiles

ALTER TABLE profiles
  ADD COLUMN IF NOT EXISTS is_sitter BOOLEAN DEFAULT FALSE,
  ADD COLUMN IF NOT EXISTS sitter_status TEXT DEFAULT NULL;

-- sitter_status può essere:
-- NULL = non ha mai fatto richiesta sitter
-- 'draft' = wizard in corso
-- 'submitted' = in attesa verifica KYD
-- 'verified' = sitter attivo
-- 'rejected' = non approvato
-- 'suspended' = sospeso

-- Aggiorna il trigger di creazione profilo
CREATE OR REPLACE FUNCTION crea_profilo_nuovo_utente()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (
    id,
    email,
    nome,
    cognome,
    is_sitter,
    sitter_status
  )
  VALUES (
    NEW.id,
    NEW.email,
    COALESCE(NEW.raw_user_meta_data->>'nome', 'Utente'),
    COALESCE(NEW.raw_user_meta_data->>'cognome', ''),
    FALSE,
    NULL
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- View aggiornata sitter completo
DROP VIEW IF EXISTS sitter_completo;
CREATE VIEW sitter_completo AS
SELECT
  sp.*,
  p.nome,
  p.cognome,
  p.nome || ' ' || LEFT(p.cognome, 1) || '.' AS nome_display,
  p.avatar_url,
  p.telefono,
  p.email,
  p.is_sitter,
  p.sitter_status
FROM sitter_profiles sp
JOIN profiles p ON sp.user_id = p.id
WHERE p.sitter_status = 'verified';
```

AGGIORNA src/hooks/useAuth.js:

```javascript
import { useState, useEffect, createContext, useContext } from 'react'
import { supabase } from '../lib/supabase'

const AuthContext = createContext({})

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null)
  const [profile, setProfile] = useState(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setUser(session?.user ?? null)
      if (session?.user) {
        caricaProfilo(session.user.id)
      } else {
        setLoading(false)
      }
    })

    const { data: { subscription } } = supabase.auth.onAuthStateChange(
      async (_event, session) => {
        setUser(session?.user ?? null)
        if (session?.user) {
          await caricaProfilo(session.user.id)
        } else {
          setProfile(null)
          setLoading(false)
        }
      }
    )

    return () => subscription.unsubscribe()
  }, [])

  const caricaProfilo = async (userId) => {
    try {
      const { data, error } = await supabase
        .from('profiles')
        .select('*')
        .eq('id', userId)
        .single()

      if (error) throw error
      setProfile(data)
    } catch (error) {
      console.error('Errore caricamento profilo:', error)
    } finally {
      setLoading(false)
    }
  }

  const registrati = async ({ email, password, nome, cognome }) => {
    const { data, error } = await supabase.auth.signUp({
      email,
      password,
      options: {
        data: { nome, cognome }
        // NESSUN ruolo fisso — tutti partono uguali
      }
    })
    if (error) throw error
    return data
  }

  const accedi = async ({ email, password }) => {
    const { data, error } = await supabase.auth.signInWithPassword({
      email,
      password
    })
    if (error) throw error
    return data
  }

  const esci = async () => {
    await supabase.auth.signOut()
    setUser(null)
    setProfile(null)
  }

  // Helpers ruolo (basati sul nuovo sistema)
  const isLoggedIn = !!user
  const isSitterVerified = profile?.sitter_status === 'verified'
  const isSitterPending = profile?.sitter_status === 'submitted'
  const isSitterDraft = profile?.sitter_status === 'draft'
  const hasSitterProfile = profile?.is_sitter === true
  const isAdmin = profile?.is_admin === true

  // Avvia il percorso "diventa sitter"
  const avviaWizardSitter = async () => {
    const { error } = await supabase
      .from('profiles')
      .update({
        is_sitter: true,
        sitter_status: 'draft'
      })
      .eq('id', user.id)

    if (error) throw error
    await caricaProfilo(user.id)
  }

  // Dopo che l'utente completa il wizard
  const submitProfiloSitter = async () => {
    const { error } = await supabase
      .from('profiles')
      .update({ sitter_status: 'submitted' })
      .eq('id', user.id)

    if (error) throw error
    await caricaProfilo(user.id)
  }

  return (
    <AuthContext.Provider value={{
      user,
      profile,
      loading,
      isLoggedIn,
      isSitterVerified,
      isSitterPending,
      isSitterDraft,
      hasSitterProfile,
      isAdmin,
      registrati,
      accedi,
      esci,
      avviaWizardSitter,
      submitProfiloSitter,
      ricaricaProfilo: () => user && caricaProfilo(user.id)
    }}>
      {children}
    </AuthContext.Provider>
  )
}

export const useAuth = () => useContext(AuthContext)
```
